Prediction Algorithm is

heartAttack <- function(cholesterol) {
    if (cholesterol < 200) 
        "no risk of heart attack" else (if (cholesterol > 199 & cholesterol < 240) 
        "Borderline high risk of heart attack" else "high risk of heart attack")
}
